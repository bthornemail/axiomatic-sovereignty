def parse_strict_yaml_front_matter(file_path):
    """
    Parses a strict subset of YAML from markdown front matter.
    Avoids heavy external YAML dependencies to keep the validator lightweight.
    """
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # Locate the front matter block delimited by ---
    match = re.match(r'^---\s*\n(.*?)\n---\s*\n', content, re.DOTALL)
    if not match:
        return None
    
    yaml_text = match.group(1)
    data = {}
    current_section = None
    
    # Basic, deterministic parsing for StrictYAML lines
    for line in yaml_text.splitlines():
        line = line.strip()
        if not line or line.startswith('#'):
            continue
        
        # Detect parent sections (e.g., logic_layer:)
        if line.endswith(':'):
            current_section = line[:-1].strip()
            data[current_section] = {}
            continue
            
        # Detect key-value pairs
        if ':' in line:
            key, val = line.split(':', 1)
            key = key.strip()
            val = val.strip().strip('"').strip("'")
            
            if current_section and line.startswith(' '):
                data[current_section][key] = val
            elif current_section and not line.startswith(' ') and current_section in data:
                # Key belongs to root if indentation drops, but simple layout assumes flat or 2-tier
                data[key] = val
            else:
                data[key] = val
                
    return data
